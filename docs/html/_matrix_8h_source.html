30 static Matrix<T> *applyComponentwise(Matrix<T> *A, Matrix<T> *B, std::function<T(T, T)> f) {
31  // check that A and B have valid dimensions:
32  // - if A and B have same dimensions -> apply f componentwise
33  // - if either A or B is a scalar -> apply scalar on each component of matrix/vector
34  bool isScalarA = A->isScalar();
35  bool isScalarB = B->isScalar();
36  if (A->getDimensions()!=B->getDimensions() && !(isScalarA ^ isScalarB)) {
37  throw std::logic_error("Matrices A and B must have the same dimensions to apply an operator componentwise, or "
38  "either one of A and B must be a scalar to apply the scalar product.");
39  } else if (isScalarA && !isScalarB) {
40  // expand A to be a (#rowsB, #columnsB)-matrix filled with the value of the scalar A
41  A->expandAndFillMatrix(B->getDimensions(), A->getScalarValue());
42  } else if (isScalarB && !isScalarA) {
43  // expand B to be a (#rowsA, #columnsA)-matrix filled with the value of the scalar B
44  B->expandAndFillMatrix(A->getDimensions(), B->getScalarValue());
45  }
46  std::vector<std::vector<T>> result;
47  for (int i = 0; i < A->values.size(); ++i) {
48  result.push_back(std::vector<T>());
49  for (int j = 0; j < A->values[0].size(); ++j) {
50  result[i].push_back(f((*A)(i, j), (*B)(i, j)));
51  }
52  }
53  return new Matrix<T>(result);
54 }

89 static Matrix<T> *applyMatrixMultiplication(Matrix<T> *matrixA, Matrix<T> *matrixB) {
90  // check that #column of matrixA equals #rows of matrixB
91  if (!matrixA->getDimensions().equals(-1, matrixB->getDimensions().numRows)) {
92  std::stringstream ss;
93  ss << "Dimension mismatch: ";
94  ss << "matrixA has dimensions " << matrixA->getDimensions() << " and matrixB has dimensions "
95  << matrixB->getDimensions() << ". ";
96  ss << "To apply matrix multiplication, #columns of matrixA must be equal to the #rows of matrixB." << std::endl;
97  throw std::logic_error(ss.str());
98  }
99  // define result matrix having dimension (#rowsA, #columnsB)
100  std::vector<std::vector<T>> result(
101  matrixA->getDimensions().numRows, std::vector<T>(matrixB->getDimensions().numColumns));
102 
103  // matrixA has dim (M,N), matrixB has dim (N,P)
104  int dimM = matrixA->getDimensions().numRows;
105  int dimN = matrixA->getDimensions().numColumns;
106  int dimP = matrixB->getDimensions().numColumns;
107  T sum = 0;
108  // Very inefficient: O(n^3)! Replace it, for example, by Strassen's multiplication algorithm.
109  for (int k = 0; k < dimN; ++k) {
110  for (int i = 0; i < dimM; ++i) {
111  for (int j = 0; j < dimP; ++j) {
112  result[i][j] += (*matrixA)(i, k)*(*matrixB)(k, j);
113  }
114  }
115  }
116  return new Matrix<T>(result);
117 }

193  void checkBoundsAndResizeMatrix(int rowIdx, int colIdx) {
194  if (rowIdx < 0 || colIdx < 0) {
195  throw std::runtime_error("Matrix access with index < 0 is invalid! "
196  "Given indices (row, col): " + std::to_string(rowIdx) + ", " + std::to_string(colIdx));
197  }
198 
199  // Resize the outer vector by adding new vectors that have the same size as the already existing ones.
200  // Note: The new size must be rowIdx+1 because a vector is 0-indexed.
201  if (rowIdx + 1 > values.size()) {
202  if (values.empty()) { // <==> values.size() == 0
203  values.resize(rowIdx + 1);
204  } else {
205  values.resize(rowIdx + 1, std::vector<T>(values.at(0).size()));
206  }
207  }
208  // Resize the inner vectors -- all of them, to ensure that all rows have the same #elements.
209  if (colIdx + 1 > values.at(rowIdx).size()) { for (auto &rw : values) rw.resize(colIdx + 1); }
210  }

318  Matrix<T> *rotate(int rotationFactor, bool inPlace) override {
319  Matrix<T> *matrixToRotate = inPlace ? this : new Matrix<T>(*this);
320  if (matrixToRotate->getDimensions().equals(1, -1)) { // a row vector
321  auto &vec = matrixToRotate->values[0];
322  std::rotate(vec.begin(), computeRotationTarget(vec.begin(), vec.size(), rotationFactor), vec.end());
323  } else if (matrixToRotate->getDimensions().equals(-1, 1)) { // a column vector
324  // Transpose the vector, rotate it, transpose it again. This is needed because std::rotate requires all elements
325  // in a single vector. As our matrix is represented using row vectors, we need to transform column->row vector.
326  matrixToRotate->transpose(true);
327  auto &vec = matrixToRotate->values[0];
328  std::rotate(vec.begin(), computeRotationTarget(vec.begin(), vec.size(), rotationFactor), vec.end());
329  matrixToRotate->transpose(true);
330  } else {
331  throw std::invalid_argument("Rotation only supported for 1-dimensional vectors.");
332  }
333  return matrixToRotate;
334  }
