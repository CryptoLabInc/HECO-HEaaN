115  template<typename R, typename T>
116  R acc(std::function<R(T, T)> func, std::vector<T> operands) {
117  auto it = operands.begin();
118  // result = operands[0] ⊕ operands[1]
119  R result = func(*it, *(++it));
120  if (operands.size() > 2) {
121  // result = ((((result ⊕ operands[2]) ⊕ operands[3]) ⊕ ...) ⊕ operands[N])
122  for (++it; it!=operands.end(); ++it) result = func(result, *it);
123  }
124  return result;
125  }

128  bool appPairwise(std::function<bool(T, T)> func, std::vector<T> operands) {
129  auto it = operands.begin();
130  // result = operands[0] ⊕ operands[1]
131  bool result = func(*it, *(++it));
132  if (operands.size() > 2) {
133  // result = ((result && (operands[1] ⊕ operands[2])) && (operands[2] ⊕ operands[3])) ...
134  for (; it!=operands.end() - 1; ++it) result = result && func(*it, *(it + 1));
135  }
136  return result;
